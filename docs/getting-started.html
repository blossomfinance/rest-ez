6link  scheme: https

7link  base_path: /content/v1

11link      method: post

12link      path: /spaces/f8bqpb154z8p/environments/master

13link      query_params:

14link        - name: access_token

15link          value: 9d5de88248563ebc0d2ad688d0473f56fcd31c600e419d6c8962f6aed0150599

16link      headers:

17link        - name: content-type

18link          value: application/json

20link        body:

21link          type: json

22link          content:

linkChained Request Flow Using Hook

Need to test complex chained API flows? Run dependencies in hooks to fetch pre-requisite data and pass it to actual test:

1linkmeta:

2link  name: Starwars suite

3linkconfiguration:

4link  scheme: https

5link  host: swapi.dev

6link  base_path: /api

7linkspecs:

8link  - name: get R2-D2 info

9link    request:

10link      path: /people/3/

11link      method: get

12link    response:

13link      status_code: 200

14link      json_data:

15link        - path: $.name

16link          value: R2-D2

17link

18link  - name: search R2-D2 info

19link    before_test:

20link      run_type: inline

21link      inline:

22link        function: !js/asyncFunction >

23link          async function() {

24link            var response = await this.runSpec('get R2-D2 info');

25link            var jsonData = JSON.parse(response.body);

26link            this.test.query_params = { name:  jsonData.name };

27link          }

28link    request:

29link      path: /people

30link      method: get

31link    response:

32link      status_code: 200

33link      custom_validator:

34link        run_type: inline

35link        inline:

36link          function: !!js/function >

37link            function() {

38link              var jsonData = JSON.parse(this.response.body);

39link              var r2d2 = jsonData.results.find(result => result.name === 'R2-D2');

40link

41link              if (!r2d2) throw new Error('R2-D2 not returned in search results');

42link            }

ℹ️ You can also place custom JS functions in a module
